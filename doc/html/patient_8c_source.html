152 patient** genererPatients(unsigned int nb_patients,temps debut_plage_horaire, temps fin_plage_horaire) {
153  /*Cette fonction permet d'instancier nb_patients patients et de leur attribuer un id et une heure d'arrivée.
154  Les heures d'arrivée sont générées au moyen d'un processus de poisson avec une forte composante aléatoire :
155  On découpe la plage horaire en paquets de 22 arrivées de patients maximum(car on devra calculer le factoriel d'un nombre, et factorielle(23) est trop grand pour le type (double))
156  On calcule ensuite à chaque instant de la plage horaire la probabilité qu'un certain nombre de patients (jusqu'à 22) soit arrivés avec la loi de poisson, en ajoutant une composante aléatoire.
157  On retient l'instant pour lequel la probabilité calculée est la plus grande.
158  La composante aléatoire est intégrée arbitrairement de manière à permettre que des patients aient une chance d'arriver en même temps et d'éviter que l'écart entre deux arrivées soit trop régulier.
159  On évite ainsi une distribution trop parfaite des patients sur la plage horaire.*/
160  unsigned int ecart_plage_horaire = diffTemps_minutes(debut_plage_horaire,fin_plage_horaire); //on calcule le nombre de minutes entre l'instant de début de la plage horaire et l'instant de fin. diffTemps est définie dans temps.c
161  patient* ppatient;//variable intermédiaire désignant le patient manipulé
162  temps heure_arrivee_generee;//Variable (temps) destinée à contenir l'heure d'arrivée générée
163  patient** liste_patients;//tableau d'éléments (patient*) qui sera renvoyé par la fonction
164  liste_patients = malloc(nb_patients*sizeof(patient*));//on réserve assez d'espace mémoire pour stocker nb_patients pointeurs vers des patients
165 
166  /* Il y a une factorielle dans le calcul de probabilite, or à partir de factorielle 22 les nombres sont trop grands pour la précision du type (double).
167  On va devoir créer des paquets de 22 patients.*/
168  double proba_occur; //probabilité d'occurence de l'événement : "un patient arrive à une minute donnée de la plage horaire". vaut nombre de patients/nombre de minutes de la plage horaire.
169  unsigned int instant_pg_proba=0;//entier qui contiendra la minute élue (plus grande probabilité calculée avec la loi de poisson)
170  double pg_proba; //réel qui contiendra la plus grande probabilité trouvée (nécessaire pour comparer les probabilités et conserver le max)
171  double proba_temp=0; //réel qui stockera les probabilités calculées avec la loi de poisson ((exp(-l*t)*(l*t)^n)/n!)
172  double aleat; //Réel pseudo aléatoire modifiant légèrement la probabilité d'occurrence utilisée dans la loi de poisson, et s'ajoutant à la probabilité calculée avec la loi de poisson elle-même
173 
174  unsigned int nb_patients_restants = nb_patients; //Nombre de patients restants après formation d'un paquet de 22 patients max
175  //toutes ces variables sont dues à la division par paquets de 22. On sépare les arrivées et la plage horaire par paquet de 22 patients pour se plier aux possibilités du C (factorielle(23) est trop grand pour le type (double)
176  unsigned int paquet_22_patients_max; //contient entre 1 et 22 patients. Nombre de patients arrivant sur la plage horaire considérée
177  unsigned int nb_paquets_22_p_max = ceil((float)nb_patients/22); //nombre de paquets formés.
178 
179  unsigned int debut_mini_plage_horaire;//La plage horaire sera subdivisée pour contenir 22 patients au maximum. On appliquera le processus de poisson à ces morceaux de plage horaires, indépendemment.
180  unsigned int fin_mini_plage_horaire=0;
181  unsigned int ecart_mini_plage_horaire;
182  unsigned int tour = 1;
183 
184  while(nb_patients_restants > 0) {
185  paquet_22_patients_max = (nb_patients_restants>22) ?22:nb_patients_restants;//à chaque tour de la boucle, on forme un nouveau paquet de 22 patients max parmis les patients restants.
186  debut_mini_plage_horaire=fin_mini_plage_horaire;//On déplace le début de la plage horaire à son ancienne fin.
187  //L'étendue de la mini plage horaire se calcule comme (((nombre patients dans le paquet)/(nombre total de patients)) * (étendue de la plage horaire complète)).
188  if(tour%2==0&&tour!=nb_paquets_22_p_max||nb_patients<=22){
189  fin_mini_plage_horaire+=ceil((float)(paquet_22_patients_max*ecart_plage_horaire)/nb_patients);//Un paquet sur deux, et pour le dernier paquet, on arrondit la fin de la plage horaire à l'entier supérieur.
190  }
191  else {
192  fin_mini_plage_horaire+=floor((float)(paquet_22_patients_max*ecart_plage_horaire)/nb_patients);//Le reste du temps, on arrondit la fin de la plage horaire à l'entier inférieur.
193  }
194 
195  ecart_mini_plage_horaire=fin_mini_plage_horaire-debut_mini_plage_horaire+1;
196  for(int p=1;p<=paquet_22_patients_max;p++){ //Pour chaque patient du paquet de 22 patients max
202  pg_proba=0;//La plus grande probabilité est initialisée à 0. Elle est toujours inférieure à au moins une probabilité générée
203 
204  for(int instant=0;instant<fin_mini_plage_horaire-debut_mini_plage_horaire;instant++) { //On va calculer proba_temp à chaque instant de la subdivision de plage horaire, pour retenir la plus grande proba et l'instant correspondant
205  //calcul de la répartition des arrivées avec une loi de poisson influencée par une composante pseudo-aléatoire
206  if(nb_patients<=22){
207  aleat=rand()/(double)RAND_MAX*(((double)nb_patients_restants/ecart_plage_horaire)/4) + 0-((double)nb_patients_restants/(ecart_plage_horaire*4)); //Pour un petit nombre de patients, on favorise arbitrairement une meilleure répartition des heures d'arrivée en diminuant le poids de la composante aléatoire.
208  }
209  else{
210  aleat=rand()/(double)RAND_MAX*(((double)nb_patients_restants/ecart_plage_horaire)/2) + 0-((double)nb_patients_restants/(ecart_plage_horaire*4));
211  }
212  proba_occur = (double)paquet_22_patients_max/ecart_mini_plage_horaire + aleat;
213 
214  //le nombre aleatoire va légèrement modifier la proba d'occurence à chaque instant ainsi que la probabilité calculée par poisson elle-même
215  proba_temp = aleat/5+(exp((double)-proba_occur*instant)*pow(((double)proba_occur*instant),p))/((double)factorielle(p)); //calcul de la probabilite que p patients soient arrives à instant avec une proba d'occurence d'arrivée de proba_occur
216  //une composante aléatoire de faible poids est ajoutée directement au résultat pour avoir une petite chance de "mauvaise répartition" voire de "collision" (deux patients ayant des heures d'arrivées proches voire identiques).
217  if(proba_temp > pg_proba) {
218  pg_proba=proba_temp; //Conservation de la plus grande probabilite (+ composante aléatoire) et de l'instant pour lequel elle est atteinte
219  instant_pg_proba=instant; //Conservation de la minute pour laquelle on a trouvé la plus grande probabilité (avec incidence de la composante aléatoire).
220  }
221  }
222  //à la fin de la boucle, instant_pg_proba est égal à la minute élue.
223 
224  heure_arrivee_generee=debut_plage_horaire; //On réinitialise l'heure d'arrivée générée au début de la plage horaire
225  for(int i=0;i<debut_mini_plage_horaire+instant_pg_proba;i++) { //generation du temps d'arrivee comme : debut_mini_plage_horaire (la première minute de la subdivision de la plage horaire) + instant_pg_proba (la minute élue);
226  incrementerTemps(&heure_arrivee_generee);
227  } // à la fin de la boucle, heure_arrivee_generee vaut : debut_plage_horaire (temps) + début subdivision de plage horaire (minutes) + minute élue (minutes)
228 
229  ppatient=creerPatient(); //instanciation d'un patient
230  ppatient->arrivee=heure_arrivee_generee;//Attribution de l'heure d'arrivée générée
231  ppatient->id=p+(nb_patients-nb_patients_restants);//p est le rang du patient dans le paquet de 22 max. nb_patients-nb_patients_restants vaut le nombre de patients déjà traités avant ce paquet.
232 
233  liste_patients[p-1+(nb_patients-nb_patients_restants)]=ppatient; //On ajoute un pointeur vers le patient généré dans le tableau qui sera retourné à la fin
234  }
235  nb_patients_restants-=paquet_22_patients_max;//Une fois tout le paquet traité, on le soustrait au total de patients à traiter
236  tour++; //On incrémente la variable tour. On reboucle et on traite un autre paquet de patients sur une autre subdivision de plage horaire.
237  }
238 
239 
240  return liste_patients;//Retour du tableau contenant les patients générés
